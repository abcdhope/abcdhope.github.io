基础server构建

本节主要是建立链接。

通过net.listen来监听端口，通过Listener.Accept()来接收客户端的请求，然后利用go来进行处理多个请求链接。

type Server struct {
	Ip   string
	Port int
}

//初始化
func NewServer(ip string, port int) *Server {
	return &Server{
		Ip:   ip,
		Port: port,
	}
}

func (s *Server) Handler(conn net.Conn) {
	fmt.Println("链接建立成功")
}

//启动服务器的接口
func (s *Server) Start() {
	//端口监听
	Listener, err := net.Listen("tcp", fmt.Sprintf("%s:%d", s.Ip, s.Port))
	if err != nil {
		fmt.Println("net.Listen err:", err)
		return
	}
	//关闭端口
	defer Listener.Close()

	for {
		//接收客户端的连接
		conn, err := Listener.Accept()
		if err != nil {
			fmt.Println("Listener err:", err)
			return
		}

		// do handler
		go s.Handler(conn)
	}

}

最后开启端口：

func main() {
	server := NewServer("127.0.0.1", 8888)
	server.Start() //开启端口
}

当客户端发送请求链接时，服务器输出链接建立成功.

用户上线功能

在上一节中，建立了服务器与客户端的连接。在这一节中，添加用户并且服务器给客户端返回消息。

首先创建用户结构，用于存放用户名字、用户地址、用户通道、用户对应的链接。

type User struct {
	Name string
	Addr string
	C    chan string //用户通道，用于将消息传给用户
	conn net.Conn    //用户链接
}

C通道用于广播消息给所有客户端，而conn链接用于对该客户端的一些操作。

然后进行初始化：

//创建一个用户的API
func NewUser(conn net.Conn, server *Server) *User {
	userAddr := conn.RemoteAddr().String() //客户端的请求地址

	user := &User{
		Name:   userAddr,
		Addr:   userAddr,
		C:      make(chan string),
		conn:   conn,
	}

	//启动监听当前user channel消息的goroutine
	go user.ListenMessage()

	return user
}

ListenMessage用于一直监听当前通道的消息，并将消息发送给给客户端:

func (u *User) ListenMessage() {
	for {
		msg := <-u.C
		u.conn.Write([]byte(msg + "\n"))
	}
}

在服务器的结构上，添加用户列表功能和消息功能，用户列表用于存放当前存在的用户，而消息功能用于将消息传递给客户端：

//创建用户列表
OnlineMap map[string]*User
mapLock   sync.RWMutex

//消息广播的channel
Message chan string

mapLock代表锁，用于当前用户的操作。

然后在handler里添加用户上线功能，并将用户添加至列表中，用于是并发运行该函数，因此加上锁来防止其他用户一起上线：

func (s *Server) Handler(conn net.Conn) {
	// fmt.Println("链接建立成功")

	user := NewUser(conn)
	//用户上线，将用户加入到OnlineMap中
	s.mapLock.Lock() //上锁
	s.OnlineMap[user.Name] = user
	s.mapLock.Unlock()
	//广播当前用户上线消息
	s.BroadCase(user, "已上线")

	//当前handler阻塞
	// select{}
}

BroadCase用于广播消息，将消息通过用户通道来传递给用户：

//广播消息
func (s *Server) BroadCase(user *User, msg string) {
	sendMsg := "[" + user.Addr + "]" + user.Name + ":" + msg
	s.Message <- sendMsg
}

且在start方法中开启一个go程来持续监听消息通道，并将通道的消息传递给用户：

//监听消息广播通道，一旦有消息则将消息广播给所有用户
func (s *Server) ListenMessage() {
	for {
		msg := <-s.Message // 无消息则阻塞
		s.mapLock.Lock()   //上锁
		for _, user := range s.OnlineMap {
			user.C <- msg //将消息传给所有用户
		}
		s.mapLock.Unlock()
	}
}

用户消息广播机制

在上一节中已经实现了通过通道来将消息传递给每个客户端，本节主要是服务器读取客户端发来的消息并将消息进行广播，核心方法还是BroadCase：

func (s *Server) Handler(conn net.Conn) {
	// fmt.Println("链接建立成功")

	user := NewUser(conn)
	//用户上线，将用户加入到OnlineMap中
	s.mapLock.Lock() //上锁
	s.OnlineMap[user.Name] = user
	s.mapLock.Unlock()
	//广播当前用户上线消息
	s.BroadCase(user, "已上线")

	//当前handler阻塞
	// select{}
	//接收客户端发送的消息并广播给所有用户
	go func() {
		buf := make([]byte, 4096)
		for {
			n, err := conn.Read(buf)
			if n == 0 {
				s.BroadCase(user, "已下线")
				return
			}
			if err != nil && err != io.EOF {
				fmt.Println("Conn Read err:", err)
			}
			msg := string(buf[:n-1])
			s.BroadCase(user, msg)
		}
	}()
}

在handler方法中增加一个go程来接收用户发来的消息，利用go程可以保持客户端与服务器互通。在go程中首先创建一个字节数组用于保存链接中的信息，并返回信息的长度，如果长度为0则表示用户断开了连接，因此给其他用户广播该用户已下线并结束这一个go程，如果长度不为0则将消息广播给所有用户，n-1表示信息里面最后一位是换行字符\n。

用户业务层封装

在上一节中，在handler中定义了用户上线和下线以及将用户发送的消息进行广播，为了方便后续增加功能，现将这些功能进行封装给用户。

首先为user结构添加server对象：

//新增Server关联
server *Server

增加的意义在于user的方法可以调用server的方法，例如消息传递功能BroadCase。

定义上线功能，功能里包含用户列表增加新用户、广播用户上线：

//用户上线
func (u *User) Online() {
	//用户上线，将用户加入到OnlineMap中
	u.server.mapLock.Lock() //上锁
	u.server.OnlineMap[u.Name] = u
	u.server.mapLock.Unlock()
	//广播当前用户上线消息
	u.server.BroadCase(u, "已上线")
}

同理定义下线功能：

//用户下线
func (u *User) Offline() {
	u.server.mapLock.Lock() //上锁
	delete(u.server.OnlineMap, u.Name)
	u.server.mapLock.Unlock()
	u.server.BroadCase(u, "已下线")
}

定义发送消息功能，将客户端发送给服务器的消息进行封装：

//用户处理消息的业务
func (u *User) DoMessage(msg string) {
	u.server.BroadCase(u, msg)
}

最后handler改为：

func (s *Server) Handler(conn net.Conn) {
	// fmt.Println("链接建立成功")

	user := NewUser(conn, s)
	//用户上线，将用户加入到OnlineMap中
	user.Online()

	//当前handler阻塞
	// select{}
	//接收客户端发送的消息并广播给所有用户
	go func() {
		buf := make([]byte, 4096)
		for {
			//读取客户端传来的数据
			n, err := conn.Read(buf)
			if n == 0 {
				user.Offline() //下线
				return
			}
			if err != nil && err != io.EOF {
				fmt.Println("Conn Read err:", err)
			}
			msg := string(buf[:n-1])
			user.DoMessage(msg) //用户针对msg进行消息处理
		}
	}()
}

在线用户查询

在上一节中定义消息处理函数DoMessage，现在在方法里增加用户查询功能，当用户输入who时，服务器返回用户在线列表：

func (u *User) DoMessage(msg string) {
	if msg == "who" {
		//查询当前在线用户都有哪些
		fmt.Println("进入该代码块")
		u.server.mapLock.Lock()
		for _, user := range u.server.OnlineMap {
			onlineMsg := "[" + user.Addr + "]" + user.Name + ":" + "在线...\n"
			u.SendMsg(onlineMsg)
		}
		u.server.mapLock.Unlock()
	} else {
		u.server.BroadCase(u, msg)
	}
}

当输入who时，遍历列表中的用户并将这些用户发送给请求的客户端，SendMsg方法用于将信息发送给当前用户：

//给当前User对应的客户端发送消息
func (u *User) SendMsg(msg string) {
	u.conn.Write([]byte(msg))
}

由于消息里存在换行符等制表符，因此调用strings.TrimSpace将主要字符的左右两端的制表符和空格进行去除：

strings.TrimSpace(s string)

修改用户名

在上一节中，定义了查询在线用户功能。在本节中通过增加修改用户名的功能来为客户端发送的修改用户名请求进行处理，消息格式定义为：rename|张三：

//用户处理消息的业务
func (u *User) DoMessage(msg string) {
	if msg == "who" {
		//查询当前在线用户都有哪些
		// fmt.Println("进入该代码块")
		u.server.mapLock.Lock()
		for _, user := range u.server.OnlineMap {
			onlineMsg := "[" + user.Addr + "]" + user.Name + ":" + "在线...\n"
			u.SendMsg(onlineMsg)
		}
		u.server.mapLock.Unlock()
	} else if len(msg) > 7 && msg[:7] == "rename|" {
		//消息格式为："rename|张三"
		//将名字进行分离出来
		newName := strings.Split(msg, "|")[1]
		if _, ok := u.server.OnlineMap[newName]; ok {
			u.SendMsg("当前用户名已被注册\n")
		} else {
			u.server.mapLock.Lock()
			//删除旧的用户
			delete(u.server.OnlineMap, u.Name)
			u.server.OnlineMap[newName] = u //将新的名字添加至用户列表中
			u.server.mapLock.Unlock()
			u.Name = newName
			u.SendMsg("您已经更新用户名：" + u.Name + "\n")
		}
	} else {
		u.server.BroadCase(u, msg)
	}
}

通过strings.Split将消息中的请求名字给提取出来，并且判断该名字是否已被注册，若被注册就给该客户端发消息：当前用户名已被注册\n，否则先删除列表中旧的用户，更新新的用户名，更新完成后给该客户端发送更新后的信息。

超时强踢功能

在上一节中，对用户名进行了修改。在本节中加入超时强踢功能，当客户端在一定时间内没有发送消息就默认掉线，服务器就关闭对该客户端的链接：

添加一个bool通道，用于判断该用户是否活跃：

//用户活跃功能
	isLive := make(chan bool)

当用户发送完消息后，将true发送给该变量：

isLive <- true //用户的任意消息表示该用户活跃

并且设置一个for循环和select用于判断该用户是否活跃，若不活跃则将用户断线：

for {
		select {
		case <-isLive:
			//当前用户活跃，重置计时器。

		case <-time.After(10 * time.Second):
			user.SendMsg("你被踢了\n")

			close(user.C) //关闭用于服务器给客户端发送消息的通道

			conn.Close() //关闭客户端发送的链接

			return
		}
	}

当isLive为true，则重置计时器。当超过一定时间，则关闭通道和链接。

私聊功能

在上一节中增加了超时强踢功能，本节增加一个私聊功能，通过”to|用户名|消息内容”来判断用户是否想开启私聊。

首先通过strings.Split来提取要对话的对象，当不存在该对象时则返回对应的消息，否则提取对话内容，并通过SendMsg方法将对话内容发送给指定的用户：

if len(msg) > 3 && msg[:3] == "to|" {
		//消息格式为："to|用户名|消息内容"
		remoteName := strings.Split(msg, "|")[1]
		//判断格式是否正确
		if remoteName == "" {
			u.SendMsg("发送内容的格式有误，正确的格式为：to|用户名|消息内容\n")
			return
		}
		//判断是否有这个人
		remoteUser, ok := u.server.OnlineMap[remoteName]
		if !ok {
			u.SendMsg("该用户名不存在\n")
			return
		}

		content := strings.Split(msg, "|")[2]
		if content == "" {
			u.SendMsg("无消息发送，请输入内容\n")
			return
		}
		//给目标用户发消息
		remoteUser.SendMsg("用户 " + u.Name + " 给您发消息：" + content + "\n")

客户端实现

在上面几节中主要是对服务器的功能进行，现在定义一个客户端来发送这些请求。

首先是定义一个结构体Client，用于存放当前用户的相关信息，例如请求的服务器的地址、端口号、用户名、用户链接、菜单选项。

//定义结构，封装功能
type Client struct {
	//服务器的Ip地址和接口
	ServerIp   string
	ServerPort int
	//用户名
	Name string
	//链接
	conn net.Conn

	//菜单选项
	flag int
}

然后利用flag标志来利用命令行输入地址和端口号：

var serverIp string
var serverPort int

//用于解析命令行的输入参数
func init() {
	// -ip 127.0.0.1 -port 9090
	flag.StringVar(&serverIp, "ip", "127.0.0.1", "设置服务器IP地址(默认为127.0.0.1)")
	flag.IntVar(&serverPort, "port", 9090, "设置服务器端口(默认为9090)")

}

func main() {
	//解析参数
	flag.Parse()
}

定义init函数来使得在主函数之前定义好地址和端口号。

然后是菜单选项，对服务器定义的一些功能进行封装，例如公聊、私聊、更新用户名等操作。

//显示用户可选择的菜单
func (c *Client) menu() bool {
	var flag int
	fmt.Println("1.公聊模式")
	fmt.Println("2.私聊模式")
	fmt.Println("3.更新用户名")
	fmt.Println("0.退出")

	fmt.Scanln(&flag)
	if flag >= 0 && flag <= 3 {
		//根据用户选择的内容来作出对应的业务
		c.flag = flag
		return true
	} else {
		fmt.Println(">>>>请输入合法范围内的数字<<<<")
		return false
	}
}

//运行业务
func (c *Client) Run() {
	for c.flag != 0 {
		//判断flag是否在可选择的范围里
		for !c.menu() {
			//循环，直到返回true
		}
		switch c.flag {
		case 1:
			//公聊模式
			c.PublicChat()
		case 2:
			//私聊模式
			c.PrivateChat()
		case 3:
			//更新用户名
			c.UpdateName()
		}
	}
	c.conn.Close()
}

用户输入对应的序号时，开启对应的功能，若用户不想进行操作，则在按0之后关闭链接下线。

封装的功能如下：

//更新用户名
func (c *Client) UpdateName() bool {
	fmt.Println(">>>>请输入用户名")

	fmt.Scanln(&c.Name) //输入需要修改的名字

	sendName := "rename|" + c.Name + "\n"
	if _, err := c.conn.Write([]byte(sendName)); err != nil {
		fmt.Println("conn write:", err)
		return false
	}
	return true
}

//公聊模式
func (c *Client) PublicChat() {
	//提示用户输入消息
	var chatMsg string
	fmt.Println("请输入消息,exit退出")
	fmt.Scanln(&chatMsg)
	//发给服务器
	for chatMsg != "exit" {
		//判断消息是否为空，不为空则发送消息
		if chatMsg != "" {
			//将消息写入链接中
			sendMsg := chatMsg + "\n"
			if _, err := c.conn.Write([]byte(sendMsg)); err != nil {
				fmt.Println("conn write err:", err)
				break
			}
		}
		//继续写入消息
		fmt.Println("请输入消息,exit退出")
		fmt.Scanln(&chatMsg)
	}
}

//查询列表中的用户
func (c *Client) SelectUsers() {
	sendMsg := "who\n"
	if _, err := c.conn.Write([]byte(sendMsg)); err != nil {
		fmt.Println("conn write err:", err)
		return
	}
}

//私聊模式
func (c *Client) PrivateChat() {
	//定义查询的名字和自定义的内容
	var remoteName string
	var chatMsg string
	//查询当前用户的列表
	c.SelectUsers()

	//输入查询的名字
	fmt.Println("请输入对话的用户名称:")
	fmt.Scanln(&remoteName)

	//输入对话内容
	fmt.Println("请输入对话内容，exit退出")
	fmt.Scanln(&chatMsg)
	for chatMsg != "exit" {
		if len(chatMsg) != 0 {
			sendMsg := "to|" + remoteName + "|" + chatMsg + "\n"
			if _, err := c.conn.Write([]byte(sendMsg)); err != nil {
				fmt.Println("conn write err:", err)
				break
			}
		}
		chatMsg = ""
		fmt.Println("请输入对话内容，exit退出")
		fmt.Scanln(&chatMsg)
	}
}

为了能够读取服务器发送的消息，定义一个方法来接收：

//接收服务器的消息
func (c *Client) DealResponse() {
	//一旦conn有数据，就直接copy到stdout标准输出上，永久阻塞监听
	if _, err := io.Copy(os.Stdout, c.conn); err != nil {
		fmt.Println("链接已关闭")
		return
	}
}

io.Copy能够将链接中的消息发送给命令行，若没有消息则进行阻塞等待消息传入。
