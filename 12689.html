Day1 database/sql 基础

通过调用标准库 database/sql来实现与数据库的交互，现在以连接SQLite数据库为例，利用github.com/mattn/go-sqlite3库来注册驱动。

sql.Open打开指定名字的数据库；Exec函数执行操作语句；RowsAffected函数返回执行操作后的信息；QueryRow函数执行查询语句，该函数与Query的区别在于它返回所有记录并一次性调用且只返回一个变量，而Query需要利用Next函数来逐步调用；Scan扫描得到的信息并写入到输入变量中，下面为一个demo：

import (
	"database/sql"
	"log"

	_ "github.com/mattn/go-sqlite3"
)

func main() {
	db, _ := sql.Open("sqlite3", "gee.db") //前者为数据库驱动名，后者为链接的数据库
	defer func() { _ = db.Close() }()      //关闭数据库
	//删除数据库中的User表
	_, _ = db.Exec("DROP TABLE IF EXISTS User;")
	//创建User表
	_, _ = db.Exec("CREATE TABLE User(Name text);")
	//注入数据
	result, err := db.Exec("INSERT INTO User('Name') values (?),(?)", "Tom", "Sam")
	if err == nil {
		//数据库更新的信息
		affected, _ := result.RowsAffected()
		log.Println(affected)
	}
	//查询数据,表示为从第1个数据进行查询
    //利用Query
	rows, _ := db.Query("SELECT Name FROM User LIMIT 1")
	//定义变量，用于输出结果
	var name string
	// if err = row.Scan(&name); err == nil {
	// 	log.Println(name)
	// }
	for rows.Next() {
		err := rows.Scan(&name)
		if err != nil {
			log.Fatal(err)
		}
		log.Println(name)
	}
}

设置输出等级，根据不同的等级来选择是否输出到终端：首先利用Iota关键字来赋值，然后创建一个函数，以等级为输入，根据输入和内置的等级来判定是否输出：

//定义层级，表示不同的等级来决定是否输出
const (
	InfoLevel = iota
	ErrorLevel
	Disabled
)

func SetLevel(level int) {
	mu.Lock()
	defer mu.Unlock()

	//先设定都输出到终端上
	for _, logger := range loggers {
		logger.SetOutput(os.Stdout)
	}
	//根据等级来决定是否输出
	if ErrorLevel < level {
		errorlog.SetOutput(ioutil.Discard) //不打印该日志
	}
	if InfoLevel < level {
		infolog.SetOutput(ioutil.Discard)
	}
}

交互数据库

定义了日志库后，以它为基础来实现go与数据库的交互，主要功能为封装sql库的几个主要函数，例如Exec、QueryRow、Query等函数，封装这几个函数需要调用sql语句来实现，因此需要定义一个函数来存放sql语句，且需要一个函数来清空之前存放的sql语句，针对以上需求来定义一个结构：

type Session struct {
	db      *sql.DB         //数据库
	sql     strings.Builder //用于拼接 SQL 语句
	sqlVars []interface{}   //SQL 语句中占位符的对应值
}

该结构包括：链接的数据库、sql语句、sql语句里占位符的值。strings.Builder的作用为写入字符串并进行存放。现在对结构进行初始化，所有的sql语句的执行都与数据库*sql.DB有关，因此只需要传入链接的数据库来初始化该结构：

func New(db *sql.DB) *Session {
	return &Session{db: db}
}

接收sql语句

定义一个方法，来作为sql语句的传入接口：

func (s *Session)Raw(sql string, value interface{}...)*Seesion{
    s.sql.WriteString(sql)
    s.sql.WriteString(" ")
    s.sqlVars=append(s.sqlVars,value...)
    return s
}

清空sql语句

定义一个方法，用于清空对象中的sql语句：

func (s *Session)Clear(){
    s.sql.Reset()
    s.sqlVars=nil
}

定义数据库接口

定义接口来接收链接的数据库，并初始化交互数据库的结构：

type Engine struct {
	db *sql.DB
}

//初始化
func NewEngine(driver, source string) (engine *Engine, err error) {
	db, err := sql.Open(driver, source)
	if err != nil {
		log.Error(err)
		return
	}

	//查看数据库是否保持连通
	if err = db.Ping(); err != nil {
		log.Error(err)
		return
	}

	engine = &Engine{db: db}
	log.Info("成功连接数据")
	return
}
//关闭数据库
func (e *Engine) Close() {
	if err := e.db.Close(); err != nil {
		log.Error("无法关闭数据库")
	}
	log.Info("成功关闭数据库")
}

//初始化数据库操作结构
func (e *Engine) NewSession() *session.Session {
	return session.New(e.db)
}

Day2 对象表结构映射

Dialect

由于SQL语句和Go语言的数据类型不同，因此需要将SQL语句和Go语言进行互相转换。而且由于不同数据库的数据类型不相同，因此可以定义一个哈希表来存储这些数据库的名字和对应的处理方式。

由于不同的数据库，其数据的处理过程不同，因此需要一个接口来定义这些处理方法，这样不同的数据库可以重写这些方法。定义Dialect接口，主要定义两个方法：1、将go语言的数据类型转换为当前数据库的数据类型DataTypeOf(typ reflect.Value) string，其中reflect.Value能够将数据类型映射；2、获取作用在表上的SQL语句TableExistSQL(tableName string) (string, []interface{})：

//数据库类型转换
type Dialect interface {
	DataTypeOf(typ reflect.Value) string                    // 将go语言的数据类型转换为数据库的数据类型
	TableExistSQL(tableName string) (string, []interface{}) //查询该表的SQL语句
}

现在定义一个哈希表用来存放对应的数据库，定义一个函数用于获取该数据库，定义一个函数用于注册数据库：

var dialectsMap = map[string]Dialect{}
//注册Dialect
func RegisterDialect(name string, dialect Dialect) {
	dialectsMap[name] = dialect
}

//获取Dialect
func GetDialect(name string) (dialect Dialect, ok bool) {
	dialect, ok = dialectsMap[name]
	return
}

由于本次使用了sqlite3数据库，因此实现该数据库的两个方法：

//定义sqlite3结构
type sqlite3 struct{}

// var _ Dialect = (*sqlite3)(nil)

//利用init函数注册sqlite3
func init() {
	RegisterDialect("sqlite3", &sqlite3{})
}

//实现方法
func (s *sqlite3) DataTypeOf(typ reflect.Value) string {
	//查找对象的类型
	switch typ.Kind() {
	case reflect.Bool:
		return "bool"
	case reflect.Int, reflect.Int8, reflect.Int16, reflect.Int32:
		return "integer"
	case reflect.Int64, reflect.Uint64:
		return "bigint"
	case reflect.Float32, reflect.Float64:
		return "real"
	case reflect.String:
		return "text"
	case reflect.Array, reflect.Slice:
		return "blob"
	case reflect.Struct:
		if _, ok := typ.Interface().(time.Time); ok {
			return "datetime"
		}
	}
	panic(fmt.Sprintf("invalid sql type %s (%s)", typ.Type().Name(), typ.Kind()))
}

func (s *sqlite3) TableExistSQL(tableName string) (string, []interface{}) {
	args := []interface{}{tableName}
	//sqlite_master是一个特殊表, 存储数据库的元信息, 如表(table), 索引(index), 视图(view), 触发器(trigger),
	//可通过select查询相关信息
	//查表
	return "SELECT name FROM sqlite_master WHERE type='table' and name=?", args
}

TableExistSQL方法返回查询特定表的SQL语句。

Schema

本节主要用于将对象转换为表，其中这两者的对应关系为：

  • 结构体名——表名
  • 结构体变量和类型——表的字段名和字段类型
  • 结构体变量的Tag——约束条件（例如非空、主键等）

以下面结构为例：

type User struct {
    Name string `geeorm:"PRIMARY KEY"`
    Age  int
}

转换成sql语句则表示为：

CREATE TABLE `User` (`Name` text PRIMARY KEY, `Age` integer);

为了能够读取变量名、类型、Tag，定义一个结构体来存储这三个值：

//字段,对应于成员变量和类型以及Tag
type Field struct {
	Name string //属性名
	Type string //属性类型
	Tag  string //tag
}

由于字段不止一个，为了方便检索和读取，定义数组和哈希表来存放字段。定义结构体来存放表：

//表的结构
type Schema struct {
	Model      interface{}
	Name       string
	Fields     []*Field
	FieldNames []string          //所有字段名
	fieldMap   map[string]*Field //字段名与字段的映射关系,通过字段名找到对应的字段
}

其中，Model表示被映射的对象（即结构体），Name表示为表名，Fields、FieldNames、fieldMap的作用是存放字段，以便于检索：

//获取字段
func (schema *Schema) GetField(name string) *Field {
	return schema.fieldMap[name]
}

现在就需要将go语言的语句转换为sqlite3的语句：

func Parse(dest interface{}, d dialect.Dialect) *Schema {
	//获取表
	modelType := reflect.Indirect(reflect.ValueOf(dest)).Type()
	schema := &Schema{
		Model:    dest,
		Name:     modelType.Name(),
		fieldMap: make(map[string]*Field),
	}
	for i := 0; i < modelType.NumField(); i++ {
		//获取第i个字段
		p := modelType.Field(i)
		//判断p是否为匿名字段或名字的首字母是否大写
		if !p.Anonymous && ast.IsExported(p.Name) {
			field := &Field{
				Name: p.Name,
				Type: d.DataTypeOf(reflect.Indirect(reflect.New(p.Type))), //利用函数对类型进行转换,reflect.New根据类型返回值
			}
			if v, ok := p.Tag.Lookup("geeorm"); ok {
				field.Tag = v
			}
			schema.Fields = append(schema.Fields, field)
			schema.FieldNames = append(schema.FieldNames, p.Name)
			schema.fieldMap[p.Name] = field
		}
	}
	return schema
}
  • 首先读取对象，得到它的结构；
  • 然后初始化Schema结构，将对象和对应的表名存放在结构中；
  • 逐一读取每一行变量、类型、Tag，其中数据类型需要转换为sqlite数据库的数据类型；
  • 将读取到的值存放在Field结构中；
  • 将多个Field对象通过之前定义的数组和哈希表来存放

针对于这个流程，使用reflect标准库来实现，这一块还需要再进一步学习。

表的操作

丰富Session结构，加入表的操作：

type Session struct {
	db       *sql.DB         //数据库
	dialect  dialect.Dialect //数据库类型转换
	refTable *schema.Schema  //表
	sql      strings.Builder //用于拼接 SQL 语句
	sqlVars  []interface{}   //SQL 语句中占位符的对应值
}

加入了数据库的标识，以及表的数据结构生成。现在定义表的生成、查询函数：

//将表存入
func (s *Session) Model(value interface{}) *Session {
	//如果没有初始化表或结构中的表与输入的表不同则重新赋值
	if s.refTable == nil || reflect.TypeOf(value) != reflect.TypeOf(s.refTable.Model) {
		s.refTable = schema.Parse(value, s.dialect)
	}
	return s
}

//判断结构中是否存放表
func (s *Session) RefTable() *schema.Schema {
	if s.refTable == nil {
		log.Error("Model is not set")
	}
	return s.refTable
}

Model函数的作用在于判断结构中的表与输入的表是否相同，避免重复写入表；RefTable判断结构中是否存放了表。

定义表的SQL语句操作函数，主要有建立表、删除表、查询表：

//创建表的SQL语句
func (s *Session) CreateTable() error {
	table := s.RefTable()
	//存放每个属性值、类型、Tag
	var columns []string
	for _, field := range table.Fields {
		columns = append(columns, fmt.Sprintf("%s %s %s", field.Name, field.Type, field.Tag))
	}
	//合并值
	desc := strings.Join(columns, ",")
	//SQL语句格式为CREATE TABLE `User` (`Name` text PRIMARY KEY, `Age` integer);
	_, err := s.Raw(fmt.Sprintf("CREATE TABLE %s (%s)", table.Name, desc)).Exec()
	return err
}

//删除表
func (s *Session) DropTable() error {
	_, err := s.Raw(fmt.Sprintf("DROP TABLE IF EXISTS %s", s.RefTable().Name)).Exec()
	return err
}

//判断数据库是否存在该表
func (s *Session) HasTable() bool {
	sql, values := s.dialect.TableExistSQL(s.RefTable().Name)
	//执行sql语句，查询表名
	row := s.Raw(sql, values...).QueryRow()
	var tmp string
	//将信息注入到tmp中
	_ = row.Scan(&tmp)
	return tmp == s.RefTable().Name
}

day3 记录新增和查询

Clause构造SQL语句

在之前的学习中，一直是通过输入SQL语句来实现对数据库的调用，而现在通过构造SQL语句来调用数据库。查询SELECT语句的构成通常如下：

SELECT col1, col2, ...
    FROM table_name
    WHERE [ conditions ]
    GROUP BY col1
    HAVING [ conditions ]

由于SELECT语句附加条件是可选的，因此需要定义专门的函数来调用这些附加条件并组合在一起。

构造结构体Clause，里面存放对应的SQL语句，然后定义函数来实现SQL语句的拼接：

type Clause struct {
	sql    map[Type]string
	sqlvar map[Type][]interface{}
}

定义数据类型Type，SQL关键字的类型都为Type，以便于统一管理：

type Type int

//定义SQL语句类型
const (
	INSERT Type = iota
	VALUES
	SELECT

定义SQL语句

由于SQL语句的生成规则不变，因此将这些规则写成私有函数，然后用哈希表来存放这些函数：

//根据SQL关键字返回对应的处理函数
//输出分别为：SQL语句，语句中的变量
type generator func(values ...interface{}) (string, []interface{})

//存放处理函数
var generators map[Type]generator

func init() {
	generators = make(map[Type]generator)
	generators[INSERT] = _insert
	generators[VALUES] = _values
	generators[SELECT] = _select
	generators[WHERE] = _where
	generators[ORDERBY] = _orderby
}

INSERT

INSERT的格式为：INSERT INTO $tableName($fields)，因此将变量嵌入到语句中并生成SQL语句：

//INSERT函数，格式为INSERT INTO $tableName($fields)
func _insert(values ...interface{}) (string, []interface{}) {
	//values第一个值为表名
	tablename := values[0]
	//传入字段名
	fields := strings.Join(values[1].([]string), ",")
	return fmt.Sprintf("INSERT INTO %s (%v)", tablename, fields), []interface{}{}
}

VALUES

VALUES的格式为：VALUES (?), (?), ...，因此只需要确定变量的个数，并生成(?)：

//生成VALUE值的个数，格式为：?, ?
func genBindVars(num int) string {
	var vars []string
	for i := 0; i < num; i++ {
		vars = append(vars, "?")
	}
	//拼接，生成?, ?
	return strings.Join(vars, ", ")
}
//VALUE函数，格式为VALUES (?), (?), ...
func _values(values ...interface{}) (string, []interface{}) {
	// VALUES ($v1), ($v2), ...
	var bindstr string //生成?
	var sql strings.Builder
	var vars []interface{}
	sql.WriteString("VALUES ")
	for i, value := range values {
		v := value.([]interface{})
		if bindstr == "" {
			bindstr = genBindVars(len(v))
		}
		sql.WriteString(fmt.Sprintf("(%v)", bindstr))
		//在(?)后面加,
		if i+1 != len(values) {
			sql.WriteString(", ")
		}
		vars = append(vars, v...)
	}
	//返回"VALUES (?), (?), ...",value1,value2
	return sql.String(), vars
}

从函数中看，由于v := value.([]interface{})通常为1个值，因此每次循环都会生成一个(?)。

SELECT

SELECT语句的格式为：SELECT $fields FROM $tableName，因此函数编写为：

//SELECT函数，格式为SELECT $fields FROM $tableName
func _select(values ...interface{}) (string, []interface{}) {
	tablename := values[0]
	fields := strings.Join(values[1].([]string), ",")
	return fmt.Sprintf("SELECT %v FROM %s", fields, tablename), []interface{}{}
}

WHERE

WHERE语句的格式为：WHERE $desc，其中desc表示条件语句：

//WHERE函数，格式为WHERE $desc，desc表示条件语句
func _where(values ...interface{}) (string, []interface{}) {
	//vars表示desc语句里的变量
	desc, vars := values[0], values[1:]
	return fmt.Sprintf("WHERE %v", desc), vars
}

ORDERBY

ORDERBY语句的格式为：ORDER BY column_name,column_name ASC|DESC;:

//ORDERBY函数，格式为ORDER BY column_name,column_name ASC|DESC;
func _orderby(values ...interface{}) (string, []interface{}) {
	return fmt.Sprintf("ORDER BY %s", values[0]), []interface{}{}
}

拼接SQL语句

当拥有这些SQL语句的生成功能后，就需要对象来调用它们，那么Clause结构体来存储这些SQL语句：

//初始化SQL语句
func (c *Clause) Set(name Type, vars ...interface{}) {
	if c.sql == nil {
		c.sql = make(map[Type]string)
		c.sqlvar = make(map[Type][]interface{})
	}
	//生成对应的sql语句
	sql, vars := generators[name](vars...)
	c.sql[name] = sql
	c.sqlvar[name] = vars
}

Clause结构体存储这些SQL语句和变量，以便于拼接：

//拼接SQL语句并输出
func (c *Clause) Build(orders ...Type) (string, []interface{}) {
	//定义变量存储SQL语句和变量值
	var sqls []string
	var vars []interface{}
	for _, order := range orders {
		if sql, ok := c.sql[order]; ok {
			//拼接sql语句
			sqls = append(sqls, sql)
			vars = append(vars, c.sqlvar[order]...)
		}
	}
	return strings.Join(sqls, " "), vars
}

按照顺序将这些SQL语句拼接得到完整SQL语句。

Insert功能

Insert 的调用方式如下：

s := geeorm.NewEngine("sqlite3", "gee.db").NewSession()
u1 := &User{Name: "Tom", Age: 18}
u2 := &User{Name: "Sam", Age: 25}
s.Insert(u1, u2, ...)

由于SQL语句需要属性值，因此结构体对象的值需要提取出来，在上一节中利用TypeOf()方法来将结构体抽象成表，现在利用ValueOf()方法来提取其中的值：在Schema结构中新增函数RecordValues：

//将结构体对象的值提取出来
func (s *Schema) RecordValues(dest interface{}) []interface{} {
	destValue := reflect.Indirect(reflect.ValueOf(dest))
	var fieldsValue []interface{}
	//根据属性名
	for _, field := range s.Fields {
		fieldsValue = append(fieldsValue, destValue.FieldByName(field.Name).Interface())
	}
	return fieldsValue
}

该函数将结构对象的值提取并用切片存储。

现在就需要生成SQL语句：

//Insert操作
func (s *Session) Insert(values ...interface{}) (int64, error) {
	//存放值
	recordValues := make([]interface{}, 0)
	for _, value := range values {
		//生成表
		table := s.Model(value).RefTable()
		//生成Insert语句
		s.clause.Set(clause.INSERT, table.Name, table.FieldNames)
		//存放值
		recordValues = append(recordValues, table.RecordValues(value))
	}
	//生成VALUES语句
	s.clause.Set(clause.INSERT, recordValues...)
	//拼接SQL语句
	sql, vars := s.clause.Build(clause.INSERT, clause.VALUES)
	result, err := s.Raw(sql, vars).Exec()
	if err != nil {
		return 0, err
	}
	return result.RowsAffected()
}

INSERT生成的格式为：

INSERT INTO table_name(col1, col2, col3, ...) VALUES
    (A1, A2, A3, ...),
    (B1, B2, B3, ...),
    ...

该函数首先生成表，然后生成INSERT语句，随后将值提取并存储在切片中，例如：[[name1,age1],[name2,age2]]，之后生成VALUES语句，有了INSERT和VALUES语句，就将它们拼接在一起，最后通过Build函数来生成最终的语句，调用Raw函数来执行该语句，利用RowsAffected函数查看表的更新情况。

实现Find功能

调用方式为:传入一个切片指针，查询的结果保存在切片中。

s := geeorm.NewEngine("sqlite3", "gee.db").NewSession()
var users []User
s.Find(&users);

利用reflect标准库来生成结构体来存放SQL查询语句得到的值：

//查询函数，空切片作为输入
func (s *Session) Find(values interface{}) error {
	destSlice := reflect.Indirect(reflect.ValueOf(values))
	destType := destSlice.Type().Elem()
	table := s.Model(reflect.New(destType).Elem().Interface()).RefTable()

	//生成SELECT语句
	s.clause.Set(clause.SELECT, table.Name, table.FieldNames)
	//sql语句
	sql, vars := s.clause.Build(clause.SELECT, clause.WHERE, clause.ORDERBY, clause.LIMIT)
	rows, err := s.Raw(sql, vars...).QueryRows()
	if err != nil {
		return err
	}
	//逐步查询
	for rows.Next() {
		//生成结构体
		dest := reflect.New(destType).Elem()
		var values []interface{}
		for _, name := range table.FieldNames {
			//我的感觉是生成空值,为了后续的写入
			values = append(values, dest.FieldByName(name).Addr().Interface())
		}
		//将SQL语句读取的值写入values
		if err := rows.Scan(values...); err != nil {
			return err
		}
		//切片中添加结构体
		destSlice.Set(reflect.Append(destSlice, dest))
	}
	return rows.Close()
}
  • 1）destSlice.Type().Elem() 获取切片的单个元素的类型 destType，使用 reflect.New() 方法创建一个 destType 的实例，作为 Model() 的入参，映射出表结构 RefTable()。
  • 2）根据表结构，使用 clause 构造出 SELECT 语句，查询到所有符合条件的记录 rows。
  • 3）遍历每一行记录，利用反射创建 destType 的实例 dest，将 dest 的所有字段平铺开，构造切片 values。
  • 4）调用 rows.Scan() 将该行记录每一列的值依次赋值给 values 中的每一个字段。
  • 5）将 dest 添加到切片 destSlice 中。循环直到所有的记录都添加到切片 destSlice 中。

链式调用

链式调用的原理为：某个对象调用某个方法后，将该对象的引用/指针返回，即可以继续调用该对象的其他方法。通常来说，当某个对象需要一次调用多个方法来设置其属性时，就非常适合改造为链式调用了。

例如SQL语句是由多个子句构成，例如SELECT语句需要设置查询条件、限制返回行数等，调用方式为：

构造条件语句函数

day5 实现钩子函数

钩子函数英文名为Hook，当代码运行到某步时，自动触发钩子函数并执行函数里的代码。

例如在ORM框架中，记录的增删查改前后设置钩子函数来执行其他内容。

实现钩子函数

GeeORM 的钩子与结构体绑定，即每个结构体需要实现各自的钩子。

//将钩子函数名称固定
const (
	BeforeQuery  = "BeforeQuery"
	AfterQuery   = "AfterQuery"
	BeforeUpdate = "BeforeUpdate"
	AfterUpdate  = "AfterUpdate"
	BeforeDelete = "BeforeDelete"
	AfterDelete  = "AfterDelete"
	BeforeInsert = "BeforeInsert"
	AfterInsert  = "AfterInsert"
)

//调用钩子函数，本质是调用结构体的方法
func (s *Session) CallMethod(method string, value interface{}) {
	//调用session里的refTable结构
	fm := reflect.ValueOf(s.RefTable().Model).MethodByName(method)
	//如果已经传入结构
	if value != nil {
		fm = reflect.ValueOf(value).MethodByName(method)
	}
	//以session作为参数
	param := []reflect.Value{reflect.ValueOf(s)}
	if fm.IsValid() {
		//执行函数，输出结果
		if v := fm.Call(param); len(v) > 0 {
			if err, ok := v[0].Interface().(error); ok {
				log.Error(err)
			}
		}
	}
}

在各个SQL方法内部调用，例如在Find方法中：

// Find gets all eligible records
func (s *Session) Find(values interface{}) error {
	s.CallMethod(BeforeQuery, nil)
    // ...
    for rows.Next() {
        dest := reflect.New(destType).Elem()
        // ...
        s.CallMethod(AfterQuery, dest.Addr().Interface())
        // ...
	}
	return rows.Close()
}

在Find方法中，在查询之前调用一次钩子函数，在查询后调用一次钩子函数，此时查询的对象中已经含有值。

Day6 支持事务

事务

数据库事务(transaction)是访问并可能操作各种数据项的一个数据库操作序列，这些操作要么全部执行,要么全部不执行，是一个不可分割的工作单位。事务由事务开始与事务结束之间执行的全部数据库操作组成。

例如A转账给B一万元，此时数据库的操作应为：

1、A的账户减去一万元；

2、B的账户增加一万元。

这两个操作全部执行表示转账成功，任何一个操作没有执行则表示转账失败，这时执行过的操作必须回退到操作之前的状态。

事务具有4个属性：ACID,即原子性（Atomicity）、一致性（Consistency）、隔离性（Isolation）、持久性（Durability）。

  • 原子性：事务中的全部操作在数据库中是不可分割的，要么全部完成，要么全部不执行。（执行所有操作）
  • 一致性：几个并行执行的事务，其执行结果必须与按某一顺序 串行执行的结果相一致。（多个事务存在顺序执行）
  • 隔离性：事务的执行不受其他事务的干扰，事务执行的中间结果对其他事务必须是透明的。（各个事务独立执行，互不干扰）
  • 持久性：对于任意已提交事务，系统必须保证该事务对数据库的改变不被丢失，即使数据库出现故障。（已完成的事务对数据库的改变不丢失）

使用以下命令来控制事务：

  • BEGIN TRANSACTION（或BEGIN）：开始事务处理。
  • COMMIT（或END TRANSACTION）：保存更改，或者可以使用 END TRANSACTION 命令。
  • ROLLBACK：回滚所做的更改。

事务只与INSERT、UPDATE 和 DELETE 以及查询操作一起使用。他们不能在创建表或删除表时使用，因为这些操作在数据库中是自动提交的。

在GO语言标准库database/sql提供了支持事务的接口，例如：

package main

import (
	"database/sql"
	_ "github.com/mattn/go-sqlite3"
	"log"
)

func main() {
	db, _ := sql.Open("sqlite3", "gee.db")//打开数据库
	defer func() { _ = db.Close() }()
	_, _ = db.Exec("CREATE TABLE IF NOT EXISTS User(`Name` text);")

	tx, _ := db.Begin()//开始事务
	_, err1 := tx.Exec("INSERT INTO User(`Name`) VALUES (?)", "Tom")
	_, err2 := tx.Exec("INSERT INTO User(`Name`) VALUES (?)", "Jack")
	if err1 != nil || err2 != nil {
		_ = tx.Rollback()//回滚
		log.Println("Rollback", err1, err2)
	} else {
		_ = tx.Commit()//提交
		log.Println("Commit")
	}
}

GeeORM 支持事务

为了能够使用事务，在session里定义Tx对象，此对象就是事务对象，由它来启动事务。此时session结构体有着两个数据库操作对象：*sql.DB和*sql.Tx，为了能够管理这两个对象，定义一个CommonDB 接口，该接口包含最基本的三个数据库操作函数：Query、QueryRow、Exec，很显然这两个对象都实现了这三个函数，因此定义一个函数，输出为该接口。这样就能选择性的输出这两个对象之一。

type Session struct {
	db       *sql.DB         //数据库
	tx       *sql.Tx         //事务选项
	dialect  dialect.Dialect //数据库类型转换
	refTable *schema.Schema  //表
	clause   clause.Clause   //SQL语句，例如INSERT、SELECT
	sql      strings.Builder //用于拼接 SQL 语句
	sqlVars  []interface{}   //SQL 语句中占位符的对应值
}
//定义接口，用于管理DB和Tx
//该接口包含了操作数据库的最常用的函数
type CommandDB interface {
	Query(query string, args ...interface{}) (*sql.Rows, error)
	QueryRow(query string, args ...interface{}) *sql.Row
	Exec(query string, args ...interface{}) (sql.Result, error)
}
//测试接口是否正常工作
var _ CommandDB = (*sql.DB)(nil)
var _ CommandDB = (*sql.Tx)(nil)

//返回数据库
func (s *Session) DB() CommandDB {
	//如果需要开始事务操作则返回事务对象
	if s.tx != nil {
		return s.tx
	}
	return s.db
}

func (s *Session) Begin() (err error) {
	log.Info("transaction begin")
	if s.tx, err = s.db.Begin(); err != nil {
		return
	}
	return
}

	if err = s.tx.Rollback(); err != nil {
		return
	}

	}

在外部接口中定义一个接口用于开始事务：

//调用事务
type TxFunc func(s *session.Session) (interface{}, error)

func (e *Engine) Transaction(f TxFunc) (result interface{}, err error) {
	s := e.NewSession()
	//开启事务
	if err := s.Begin(); err != nil {
		return nil, err
	}
	//defer操作，防止函数出现错误以进行回调
	defer func() {
		if p := recover(); p != nil {
			_ = s.RollBack() //回滚
			panic(p)         //回滚后再报错，确保数据库没有被改变
		} else {
			//为了防止提交出问题或是在操作执行中出问题时进行回滚
			defer func() {
				if err != nil {
					_ = s.RollBack()
				}
			}()
			//若err不为空则表示函数执行过程中出错
			if err == nil {
				err = s.Commit()
			}
		}
	}()
	return f(s)
}

事务的所有操作封装成一个函数TxFunc，执行该函数等于在完成事务，利用defer延迟执行函数完成后的工作，执行函数时可能存在三种：1、函数的代码有问题；2、操作执行的过程中出问题；3、函数执行完成。因此使用判断语句来选择是否回滚。

day7 数据库迁移

  • 结构体(struct)变更时，数据库表的字段(field)自动迁移(migrate)。
  • 仅支持字段新增与删除，不支持字段类型变更。

使用SQL语句Migrate

数据库迁移存在着很多情况，复杂的情况表示为外键关系等关联关系，简单的情况表示为字段的增加和删除。

在本节中，主要实现字段的增加和删除。

新增字段

新增字段的语法为：

ALTER TABLE table_name ADD COLUMN col_name, col_type;

删除字段

删除字段的一个可行方法为新建一个表，从旧的表中选择需要的字段，然后删除旧的表：

CREATE TABLE new_table AS SELECT col1, col2, ... from old_table
DROP TABLE old_table
ALTER TABLE new_table RENAME TO old_table;

Go实现

利用事务来实现这两个操作，输入一个结构体来进行判断是否新增或删除字段：

//输出两个字符数组中不同的元素
func difference(a []string, b []string) (diff []string) {
	//利用哈希表
	mapB := make(map[string]bool)
	for _, v := range b {
		mapB[v] = true
	}
	for _, v := range a {
		//查找a中有而b中没有的元素
		if _, ok := mapB[v]; !ok {
			diff = append(diff, v)
		}
	}
	return
}

func (e *Engine) Migrate(value interface{}) error {
	//启用事务
	_, err := e.Transaction(func(s *session.Session) (result interface{}, err error) {
		//判断数据库中是否存在该表
		if !s.Model(value).HasTable() {
			//创建表
			log.Infof("table %s doesn't exist", s.RefTable().Name)
			return nil, s.CreateTable()
		}
		//获取表的信息
		table := s.RefTable()
		rows, _ := s.Raw(fmt.Sprintf("SELECT * FROM %s LIMIT 1", table.Name)).QueryRows()
		//从rows获取属性
		columns, _ := rows.Columns()
		//比较不同
		addCols := difference(table.FieldNames, columns)
		delCols := difference(columns, table.FieldNames)
		log.Infof("added cols %v, deleted cols %v", addCols, delCols)

		//增加字段
		for _, col := range addCols {
			//根据字段名获取属性信息
			field := table.GetField(col)
			//更新
			sqlStr := fmt.Sprintf("ALTER TABLE %s ADD COLUMN %s %s", table.Name, field.Name, field.Type)
			//执行SQL语句
			if _, err = s.Raw(sqlStr).Exec(); err != nil {
				return
			}
		}
		//删除字段
		if len(delCols) == 0 {
			return
		}
		//创建新表
		tmp := "tmp_" + table.Name
		//拼接属性名
		fieldStr := strings.Join(table.FieldNames, ", ")
		/*
			CREATE TABLE new_table AS SELECT col1, col2, ... from old_table
			DROP TABLE old_table
			ALTER TABLE new_table RENAME TO old_table;
		*/
		s.Raw(fmt.Sprintf("CREATE TABLE %s AS SELECT %s from %s;", tmp, fieldStr, table.Name))
		s.Raw(fmt.Sprintf("DROP TABLE %s;", table.Name))
		s.Raw(fmt.Sprintf("ALTER TABLE %s RENAME TO %s;", table.Name, tmp))
		_, err = s.Exec()
		return
	})
	return err
}

该代码的流程为：

  • 判断是否存在该表，不存在则创建表并返回；
  • 获取该表的信息，并得到它的属性名；
  • 判断输入结构的属性名与已有的表的属性的差异（存在输入的结构与之前的表的名字相同但属性不同，因此需要更新）；
  • 新增字段；
  • 删除字段；
